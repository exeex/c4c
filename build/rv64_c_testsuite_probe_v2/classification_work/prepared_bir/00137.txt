bir.func @main() -> i32 {
entry:

  @.str0 = bir.load_local ptr %lv.p, addr .str0
  bir.store_local %lv.p, ptr @.str0
  %t1 = bir.load_local ptr %lv.p
  %t2 = bir.load_local i8 %t2.addr, addr %t1
  %t3 = bir.sext i8 %t2 to i32
  %t4 = bir.eq i32 %t3, 104
  bir.cond_br i32 %t4, tern.then.7, tern.else.9
tern.then.7:
  bir.br tern.then.end.8
tern.then.end.8:
  bir.br tern.end.11
tern.else.9:
  bir.br tern.else.end.10
tern.else.end.10:
  bir.br tern.end.11
tern.end.11:
  %t12 = bir.select eq i32 %t3, 104, i32 0, 1
  bir.ret i32 %t12
}
